CREATE TABLE IF NOT EXISTS leads (
  id INTEGER PRIMARY KEY, name TEXT, erp TEXT,
  branche TEXT, land TEXT, size TEXT,
  stage TEXT DEFAULT 'Lead erkannt',
  assign TEXT DEFAULT '', notes_history JSONB DEFAULT '[]',
  win_chance INTEGER DEFAULT 50,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE leads DISABLE ROW LEVEL SECURITY;
GRANT ALL ON leads TO anon, authenticated;
